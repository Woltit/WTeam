enum Role {
  ADMIN
  MODER
  USER
}

enum ItemCondition {
  IDEAL
  GOOD
  NORM
  BAD
  NEEDS_REPAIRING
}

enum RentingStatus {
  ACTIVE
  RENTED
  INACTIVE
  DELETED
}

enum VerificationStatus {
  PENDING
  VERIFIED
  REJECTED
}

TABLE users {
  id BIGSERIAL [pk]
  email VARCHAR(255) [unique, not null]
  password VARCHAR(255)
  role Role [not null, default: 'USER']

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp [not null, default: `now()`]
}

TABLE user_profiles {
  id BIGSERIAL [pk]
  user_id BIGINT [ref: - users.id ]
  last_name VARCHAR(100) [not |null]
  first_name VARCHAR(100) [not null]
  middle_name VARCHAR(100)
  birth_date DATE [not null]
  phone VARCHAR(15) [not null]
  bio TEXT
  passport_id VARCHAR(50) [unique, not null]
  verification_status VerificationStatus [not null, default: 'PENDING']

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp [not null, default: `now()`]
}

TABLE user_reviews {
  id BIGSERIAL [pk]
  user_id BIGINT [ref: > users.id]
  reviewer_id BIGINT [ref: > users.id]
  trust_rating SMALLINT [not null, note: '1-5']
  comment TEXT

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp [not null, default: `now()`]
}

TABLE categories {
  id BIGSERIAL [pk]
  name VARCHAR(255) [not null]
}

TABLE items {
  id BIGSERIAL [pk]
  owner_id BIGINT [not null, ref: > users.id]
  category_id BIGINT [not null, ref: > categories.id]
  title VARCHAR(100) [not null]
  description TEXT
  condition ItemCondition [not null]
  price_per_day decimal(10, 2) [not null]
  deposit_amount_per_day decimal(10, 2) [not null]

  status RentingStatus [not null]
  city VARCHAR(100) [not null]
  address TEXT [not null]
  is_verified bool [not null, default: false]

  created_at timestamp [not null, default: `now()`]
  updated_at timestamp [not null, default: `now()`]

enum BookingStatus
  PENDING      // Очікує підтвердження власником
  APPROVED     // Підтверджено власником
  REJECTED     // Відхилено власником
  PAID         // Оплачено орендарем
  IN_PROGRESS  // Річ у орендаря (активна оренда)
  COMPLETED    // Річ повернута
  CANCELLED    // Скасовано
  DISPUTE      // Відкрито суперечку

TABLE bookings {
  id BIGSERIAL [pk]
  item_id BIGINT [ref: > items.id]
  renter_id BIGINT [ref: > users.id]
  start_date DATE [not null]
  end_date DATE [not null]
  total_price decimal(10, 2) [not null]
  deposit_total decimal(10, 2) [not null]
  status BookingStatus [not null, default: 'PENDING']

  created_at timestamp [default: `now()`]
  updated_at timestamp [default: `now()`]
}

enum TransactionType {
  RENT_PAYMENT
  DEPOSIT_HOLD
  DEPOSIT_REFUND
  COMPENSATION
}

TABLE transactions {
  id BIGSERIAL [pk]
  booking_id BIGINT [ref: > bookings.id]
  amount decimal(10, 2) [not null]
  type TransactionType [not null]
  external_tx_id VARCHAR(255) // ID транзакції з платіжної системи (Stripe/Mono/WayForPay)
  status VARCHAR(50) // success, pending, failed

  created_at timestamp [default: `now()`]
}

TABLE item_images {
  id BIGSERIAL [pk]
  item_id BIGINT [ref: > items.id]
  image_url TEXT [not null]
  is_main BOOLEAN [default: false]
  created_at timestamp [default: `now()`]
}

TABLE item_reviews {
  id BIGSERIAL [pk]
  item_id BIGINT [ref: > items.id]
  renter_id BIGINT [ref: > users.id]
  rating SMALLINT [not null, note: '1-5']
  comment TEXT
  created_at timestamp [default: `now()`]
}

TABLE chat_rooms {
  id BIGSERIAL [pk]
  booking_id BIGINT [ref: - bookings.id] // Опційно: прив'язка до конкретного замовлення
  created_at timestamp [default: `now()`]
}

TABLE messages {
  id BIGSERIAL [pk]
  room_id BIGINT [ref: > chat_rooms.id]
  sender_id BIGINT [ref: > users.id]
  message_text TEXT [not null]
  is_read BOOLEAN [default: false]
  created_at timestamp [default: `now()`]
}
